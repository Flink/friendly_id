def error_message
  "Couldn't find all %s with IDs (%s) AND %s (found %d results, but was looking for %d)" % [
    model.name.pluralize,
    ids.join(', '),
    sanitize_sql(options[:conditions]),
    results.size,
    expected_size
  ]
end
